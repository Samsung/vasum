101 int Socket::getSystemdSocket(const std::string& path)
102 {
103  int n = ::sd_listen_fds(-1 /*Block further calls to sd_listen_fds*/);
104  if (n < 0) {
105  LOGE("sd_listen_fds fails with errno: " << n);
106  throw IPCException("sd_listen_fds fails with errno: " + std::to_string(n));
107  }
108 
109  for (int fd = SD_LISTEN_FDS_START;
110  fd < SD_LISTEN_FDS_START + n;
111  ++fd) {
112  if (0 < ::sd_is_socket_unix(SD_LISTEN_FDS_START, SOCK_STREAM, 1, path.c_str(), 0)) {
113  return fd;
114  }
115  }
116  LOGW("No usable sockets were passed by systemd.");
117  return -1;
