476 bool Processor::onMethodRequest(MethodRequest& request)
477 {
478  LOGS(mLogPrefix + "Processor onMethodRequest");
479 
480  auto peerIt = getPeerInfoIterator(request.peerID);
481 
482  if (peerIt == mPeerInfo.end()) {
483  LOGE(mLogPrefix + "Peer disconnected. No user with a peerID: " << request.peerID);
484 
485  // Pass the error to the processing callback
486  ResultBuilder resultBuilder(std::make_exception_ptr(IPCPeerDisconnectedException()));
487  IGNORE_EXCEPTIONS(request.process(resultBuilder));
488 
489  return false;
490  }
491 
492  if (mReturnCallbacks.count(request.messageID) != 0) {
493  LOGE(mLogPrefix + "There already was a return callback for messageID: " << request.messageID);
494  }
495  mReturnCallbacks[request.messageID] = std::move(ReturnCallbacks(peerIt->peerID,
496  std::move(request.parse),
497  std::move(request.process)));
498 
499  try {
500  // Send the call with the socket
501  Socket& socket = *peerIt->socketPtr;
502  Socket::Guard guard = socket.getGuard();
503  socket.write(&request.methodID, sizeof(request.methodID));
504  socket.write(&request.messageID, sizeof(request.messageID));
505  LOGT(mLogPrefix + "Serializing the message");
506  request.serialize(socket.getFD(), request.data);
507  } catch (const std::exception& e) {
508  LOGE(mLogPrefix + "Error during sending a method: " << e.what());
509 
510  // Inform about the error
511  ResultBuilder resultBuilder(std::make_exception_ptr(IPCSerializationException()));
512  IGNORE_EXCEPTIONS(mReturnCallbacks[request.messageID].process(resultBuilder));
513 
514 
515  mReturnCallbacks.erase(request.messageID);
516  removePeerInternal(peerIt,
517  std::make_exception_ptr(IPCSerializationException()));
518  return true;
519 
520  }
521 
522  return false;
523 }

649 bool Processor::onFinishRequest(FinishRequest& request)
650 {
651  LOGS(mLogPrefix + "Processor onFinishRequest");
652 
653  // Clean the mRequestQueue
654  while (!mRequestQueue.isEmpty()) {
655  auto request = mRequestQueue.pop();
656  LOGE(mLogPrefix + "Got: " << request.requestID << " after FINISH");
657 
658  switch (request.requestID) {
659  case Event::METHOD: {
660  auto requestPtr = request.get<MethodRequest>();
661  ResultBuilder resultBuilder(std::make_exception_ptr(IPCClosingException()));
662  IGNORE_EXCEPTIONS(requestPtr->process(resultBuilder));
663  break;
664  }
665  case Event::REMOVE_PEER: {
666  onRemovePeerRequest(*request.get<RemovePeerRequest>());
667  break;
668  }
669  case Event::SEND_RESULT: {
670  onSendResultRequest(*request.get<SendResultRequest>());
671  break;
672  }
673  case Event::SIGNAL:
674  case Event::ADD_PEER:
675  case Event::FINISH:
676  break;
677  }
678  }
679 
680  // Close peers
681  while (!mPeerInfo.empty()) {
682  removePeerInternal(--mPeerInfo.end(),
683  std::make_exception_ptr(IPCClosingException()));
684  }
685 
686  mIsRunning = false;
687 
688  request.conditionPtr->notify_all();
689  return true;
690 }
