47 void waitForEvent(int fd,
48  short event,
49  const chr::high_resolution_clock::time_point deadline)
50 {
51  // Wait for the rest of the data
52  struct pollfd fds[1];
53  fds[0].fd = fd;
54  fds[0].events = event;
55 
56  for (;;) {
57  chr::milliseconds timeoutMS = chr::duration_cast<chr::milliseconds>(deadline - chr::high_resolution_clock::now());
58  if (timeoutMS.count() < 0) {
59  LOGE("Timeout while waiting for event: " << std::hex << event <<
60  " on fd: " << std::dec << fd);
61  throw UtilsException("Timeout");
62  }
63 
64  int ret = ::poll(fds, 1 /*fds size*/, timeoutMS.count());
65 
66  if (ret == -1) {
67  if (errno == EINTR) {
68  continue;
69  }
70  LOGE("Error in poll: " + getSystemErrorMessage());
71  throw UtilsException("Error in poll: " + getSystemErrorMessage());
72  }
73 
74  if (ret == 0) {
75  LOGE("Timeout in read");
76  throw UtilsException("Timeout in read");
77  }
78 
79  if (fds[0].revents & event) {
80  // Here Comes the Sun
81  break;
82  }
83 
84  if (fds[0].revents & POLLHUP) {
85  LOGW("Peer disconnected");
86  throw UtilsException("Peer disconnected");
87  }
88  }
89 }

111 void write(int fd, const void* bufferPtr, const size_t size, int timeoutMS)
112 {
113  chr::high_resolution_clock::time_point deadline = chr::high_resolution_clock::now() +
114  chr::milliseconds(timeoutMS);
115 
116  size_t nTotal = 0;
117  for (;;) {
118  int n = ::write(fd,
119  reinterpret_cast<const char*>(bufferPtr) + nTotal,
120  size - nTotal);
121  if (n >= 0) {
122  nTotal += n;
123  if (nTotal == size) {
124  // All data is written, break loop
125  break;
126  }
127  } else if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
128  // Neglected errors
129  LOGD("Retrying write");
130  } else {
131  LOGE("Error during writing: " + getSystemErrorMessage());
132  throw UtilsException("Error during writing: " + getSystemErrorMessage());
133  }
134 
135  waitForEvent(fd, POLLOUT, deadline);
136  }
