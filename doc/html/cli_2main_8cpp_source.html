271 {
272  out << "Usage: " << name << " [command [-h|args]]\n\n"
273  << "command can be one of the following:\n";

282 int main(const int argc, const char** argv)
283 {
284  if (argc < 2) {
285  printUsage(std::cout, argv[0]);
286  return EXIT_FAILURE;
287  }
288  if (commands.count(argv[1]) == 0) {
289  printUsage(std::cout, argv[0]);
290  return EXIT_FAILURE;
291  }
292 
293  CommandLineInterface& command = commands[argv[1]];
294  auto it = std::find(argv, argv+argc, std::string("-h"));
295  if (it != argv + argc) {
296  command.printUsage(std::cout);
297  return EXIT_SUCCESS;
298  }
299 
300  try {
301  command.execute(1, argc, argv);
302  } catch (const std::runtime_error& ex) {
303  std::cerr << ex.what() << std::endl;
304  return EXIT_FAILURE;
305  }
306  return EXIT_SUCCESS;
307 }
