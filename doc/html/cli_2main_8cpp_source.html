336 void printUsage(std::ostream& out, const std::string& name, unsigned int mode)
337 {
338  std::string n;
339  if (!name.empty()) {

342 
343  if (mode == MODE_COMMAND_LINE) {
344  out << "Usage: " << n << "[-h|-f <filename>|[<command> [-h|<args>]]\n"
345  << "Called without parameters enters interactive mode.\n"
346  << "Options:\n"
347  << "-h print help\n"
348  << "-f <filename> read and execute commands from file\n\n";
349  } else {
350  out << "Usage: [-h|<command> [-h|<args>]]\n\n";

352  out << "command can be one of the following:\n";
353

391  if (pair == commands.end()) {
392  return EXIT_FAILURE;

396  if (!command.isAvailable(mode)) {

399 
400  auto it = std::find(argv.begin(), argv.end(), std::string("-h"));
401  if (it != argv.end()) {

418 char* object_generator(const char* text, int state)
419 {
420  static std::vector<std::string> objs;
421  static size_t len = 0, index = 0;
422 
423  if (state == 0) {
424  objs.clear();
425  len = ::strlen(text);
426  index = 0;
427 
428  using namespace std::placeholders;
429  VsmArrayString ids = NULL;
430  try {
431  CommandLineInterface::executeCallback(bind(vsm_get_zone_ids, _1, &ids));
432  } catch (const std::runtime_error& ex) {
433  // Quietly ignore, nothing we can do anyway
434  }
435 
436  if (ids != NULL) {
437  for (VsmString* id = ids; *id; ++id) {
438  if (::strncmp(text, *id, len) == 0) {
439  objs.push_back(*id);
440  }
441  }
442 
443  vsm_array_string_free(ids);
444  }
445  }
446 
447  if (index < objs.size()) {
448  return ::strdup(objs[index++].c_str());
449  }
450 
451  return NULL;
452 }

454 char* cmd_generator(const char* text, int state)
455 {
456  static std::vector<std::string> cmds;
457  static size_t len = 0, index = 0;
458 
459  if (state == 0) {
460  cmds.clear();
461  len = ::strlen(text);
462  index = 0;
463 
464  for (const auto& command : commands) {
465  if (command.second.isAvailable(MODE_INTERACTIVE)) {
466  const std::string& cmd = command.second.getName();
467  if (::strncmp(text, cmd.c_str(), len) == 0) {
468  cmds.push_back(cmd);
469  }
470  }
471  }
472  }
473 
474  if (index < cmds.size()) {
475  return ::strdup(cmds[index++].c_str());
476  }
477 
478  return NULL;

494 static bool readline_from(const std::string& prompt, std::istream& stream, std::string& ln)
495 {
496  if (interactiveMode) {
497  char *cmd = ::readline(prompt.c_str());
498  if (cmd == NULL) {
499  return false;
500  }
501  ln = cmd;
502  free(cmd);
503  } else {
504  std::getline(stream, ln);
505  if (!stream.good()) {
506  return false;
507  }
508  }
509 
510  if (interactiveMode && !ln.empty()) {
511  ::add_history(ln.c_str());
512  }
513 
514  return true;
515 }

517 static int processStream(std::istream& stream)
518 {
519  if (connect() != EXIT_SUCCESS) {
520  return EXIT_FAILURE;
521  }
522 
523  std::string ln;
524  while (readline_from(">>> ", stream, ln)) {
525  if (ln.empty() || ln[0] == '#') { //skip empty line or comment
526  continue;
527  }
528 
529  std::istringstream iss(ln);
530  Args argv{std::istream_iterator<std::string>{iss},
531  std::istream_iterator<std::string>{}};
532 
533  if (commands.count(argv[0]) == 0) {
534  printUsage(std::cout, "", MODE_INTERACTIVE);
535  continue;
536  }
537 
538  executeCommand(argv, MODE_INTERACTIVE);
539  }
540 
541  disconnect();
542  return EXIT_SUCCESS;
543 }
