55 void one_shot(const function<VsmStatus(VsmClient)>& fun)
56 {
57  string msg;
58  VsmStatus status;
59  VsmClient client;
60 
61  status = vsm_start_glib_loop();
62  if (VSMCLIENT_SUCCESS != status) {
63  throw runtime_error("Can't start glib loop");
64  }
65 
66  client = vsm_client_create();
67  if (NULL == client) {
68  msg = "Can't create client";
69  goto finish;
70  }
71 
72  status = vsm_connect(client);
73  if (VSMCLIENT_SUCCESS != status) {
74  msg = vsm_get_status_message(client);
75  goto finish;
76  }
77 
78  status = fun(client);
79  if (VSMCLIENT_SUCCESS != status) {
80  msg = vsm_get_status_message(client);
81  goto finish;
82  }
83 
84 finish:
85  vsm_client_free(client);
86  vsm_stop_glib_loop();
87  if (! msg.empty()) {
88  throw runtime_error(msg);
89  }
90 }
