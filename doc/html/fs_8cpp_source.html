60 bool readFileContent(const std::string& path, std::string& result)
61 {
62  std::ifstream file(path);
63 
64  if (!file) {
65  LOGD(path << ": could not open for reading");
66  return false;
67  }
68 
69  file.seekg(0, std::ios::end);
70  std::streampos length = file.tellg();
71  if (length < 0) {
72  LOGD(path << ": tellg failed");
73  return false;
74  }
75  result.resize(static_cast<size_t>(length));
76  file.seekg(0, std::ios::beg);
77 
78  file.read(&result[0], length);
79  if (!file) {
80  LOGD(path << ": read error");
81  result.clear();
82  return false;
83  }
84 
85  return true;
86 }

266 bool copyDirContentsRec(const boost::filesystem::path& src, const boost::filesystem::path& dst)
267 {
268  try {
269  for (fs::directory_iterator file(src);
270  file != fs::directory_iterator();
271  ++file) {
272  fs::path current(file->path());
273  fs::path destination = dst / current.filename();
274 
275  boost::system::error_code ec;
276 
277  if (!fs::is_symlink(current) && fs::is_directory(current)) {
278  fs::create_directory(destination, ec);
279  } else {
280  fs::copy(current, destination, ec);
281  }
282 
283  if (ec.value() != boost::system::errc::success) {
284  LOGW("Failed to copy " << current << ": " << ec.message());
285  continue;
286  }
287 
288  if (!fs::is_symlink(current) && fs::is_directory(current)) {
289  if (!copyDirContentsRec(current, destination)) {
290  return false;
291  }
292 
293  // apply permissions coming from source file/directory
294  fs::file_status stat = status(current);
295  fs::permissions(destination, stat.permissions(), ec);
296 
297  if (ec.value() != boost::system::errc::success) {
298  LOGW("Failed to set permissions for " << destination << ": " << ec.message());
299  }
300  }
301 
302  // change owner
303  struct stat info;
304  ::stat(current.string().c_str(), &info);
305  if (fs::is_symlink(destination)) {
306  if (::lchown(destination.string().c_str(), info.st_uid, info.st_gid) < 0) {
307  LOGW("Failed to change owner of symlink " << destination.string() << ": " << getSystemErrorMessage());
308  }
309  } else {
310  if (::chown(destination.string().c_str(), info.st_uid, info.st_gid) < 0) {
311  LOGW("Failed to change owner of file " << destination.string() << ": " << getSystemErrorMessage());
312  }
313  }
314  }
315  } catch (fs::filesystem_error& e) {
316  LOGW(e.what());
317  }
318 
319  return true;
320 }

341 bool createDir(const std::string& path, uid_t uid, uid_t gid, boost::filesystem::perms mode)
342 {
343  fs::path dirPath(path);
344  boost::system::error_code errorCode;
345  bool runDirCreated = false;
346  if (!fs::exists(dirPath)) {
347  if (!fs::create_directory(dirPath, errorCode)) {
348  LOGE("Failed to create directory '" << path << "': "
349  << errorCode.message());
350  return false;
351  }
352  runDirCreated = true;
353  } else if (!fs::is_directory(dirPath)) {
354  LOGE("Path '" << path << " already exists");
355  return false;
356  }
357 
358  // set permissions
359  fs::permissions(dirPath, mode, errorCode);
360  if (fs::status(dirPath).permissions() != mode) {
361  LOGE("Failed to set permissions to '" << path << "': "
362  << errorCode.message());
363  return false;
364  }
365 
366  // set owner
367  if (::chown(path.c_str(), uid, gid) != 0) {
368  int err = errno;
369  // remove the directory only if it hadn't existed before
370  if (runDirCreated) {
371  fs::remove(dirPath);
372  }
373  LOGE("chown() failed for path '" << path << "': " << getSystemErrorMessage(err));
374  return false;
375  }
376 
377  return true;
378 }

380 bool createDirs(const std::string& path, mode_t mode)
381 {
382  const boost::filesystem::perms perms = getPerms(mode);
383  std::vector<fs::path> dirsCreated;
384  fs::path prefix;
385  const fs::path dirPath = fs::path(path);
386  for (const auto& dirSegment : dirPath) {
387  prefix /= dirSegment;
388  if (!fs::exists(prefix)) {
389  bool created = createDir(prefix.string(), -1, -1, perms);
390  if (created) {
391  dirsCreated.push_back(prefix);
392  } else {
393  LOGE("Failed to create dir");
394  // undo
395  for (auto iter = dirsCreated.rbegin(); iter != dirsCreated.rend(); ++iter) {
396  boost::system::error_code errorCode;
397  fs::remove(*iter, errorCode);
398  if (errorCode) {
399  LOGE("Error during cleaning: dir: " << *iter
400  << ", msg: " << errorCode.message());
401  }
402  }
403  return false;
404  }
405  }
406  }
