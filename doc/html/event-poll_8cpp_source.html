104 bool EventPoll::dispatchIteration(const int timeoutMs)
105 
106 {
107  for (;;) {
108  epoll_event event;
109  int num = epoll_wait(mPollFD, &event, 1, timeoutMs);
110  if (num == 0) {
111  return false; // timeout
112  }
113  if (num < 0) {
114  if (errno == EINTR) {
115  continue;
116  }
117  LOGE("Failed to wait on epoll: " << getSystemErrorMessage());
118  throw UtilsException("Could not wait for event");
119  }
120 
121  // callback could be removed in the meantime, so be careful, find it inside lock
122  std::lock_guard<Mutex> lock(mMutex);
123  auto iter = mCallbacks.find(event.data.fd);
124  if (iter == mCallbacks.end()) {
125  continue;
126  }
127 
128  // add ref because removeFD(self) can be called inside callback
129  std::shared_ptr<Callback> callback(iter->second);
130  try {
131  LOGT("Dispatch fd: " << event.data.fd << ", events: " << eventsToString(event.events));
132  (*callback)(event.data.fd, event.events);
133  return true;
134  } catch (std::exception& e) {
135  LOGE("Got unexpected exception: " << e.what());
136  assert(0 && "Callback should not throw any exceptions");
137  return true;
138  }
139  }
140 }

143 {
