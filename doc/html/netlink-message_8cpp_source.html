291 NetlinkResponse send(const NetlinkMessage& msg, int pid)
292 {
293  assert(msg.hdr().nlmsg_flags & NLM_F_ACK);
294 
295  std::unique_ptr<std::vector<char>> data;
296  Netlink nl;
297  nl.open(pid);
298  try {
299  nl.send(&msg.hdr());

301  } catch (const std::exception& ex) {

303  nl.close();

305  }
