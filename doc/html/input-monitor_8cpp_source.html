113 bool isDeviceWithName(const boost::regex& deviceNameRegex,
114  const fs::directory_entry& directoryEntry)
115 {
116  std::string path = directoryEntry.path().string();
117 
118  if (!utils::isCharDevice(path)) {
119  return false;
120  }
121 
122  int fd = open(path.c_str(), O_RDONLY);
123  if (fd < 0) {
124  LOGD("Failed to open " << path);
125  return false;
126  }
127 
128  char name[DEVICE_NAME_LENGTH];
129  if (ioctl(fd, EVIOCGNAME(sizeof(name)), name) < 0) {
130  LOGD("Failed to get the device name of: " << path);
131  if (close(fd) < 0) {
132  LOGE("Error during closing file " << path);
133  }
134  return false;
135  }
136 
137  if (close(fd) < 0) {
138  LOGE("Error during closing file " << path);
139  }
140 
141  LOGD("Checking device: " << name);
142  if (boost::regex_match(name, deviceNameRegex)) {
143  LOGI("Device file found under: " << path);
144  return true;
145  }
146 
147  return false;
148 }

151 std::string InputMonitor::getDevicePath() const
152 {
153  std::string device = mConfig.device;
154  if (fs::path(device).is_absolute()
155  && fs::exists(device)) {
156  LOGD("Device file path is given");
157  return device;
158  }
159 
160  // device name is given - device file path is to be determined
161  LOGT("Determining, which device node is assigned to '" << device << "'");
162  using namespace std::placeholders;
163  fs::directory_iterator end;
164  boost::regex deviceNameRegex(".*" + device + ".*");
165  const auto it = std::find_if(fs::directory_iterator(DEVICE_DIR),
166  end,
167  std::bind(isDeviceWithName, deviceNameRegex, _1));
168  if (it == end) {
169  LOGE("None of the files under '" << DEVICE_DIR <<
170  "' represents device named: " << device);
171  throw InputMonitorException("Cannot find a device");
172  }
173 
174  return it->path().string();
175 }

179 void InputMonitor::createGIOChannel(const std::string& devicePath)
180 {
181  // We need NONBLOCK for FIFOs in the tests
182  int fd = open(devicePath.c_str(), O_RDONLY | O_NONBLOCK);
183  if (fd < 0) {
184  LOGE("Cannot create input monitor channel. Device file: " <<
185  devicePath << " doesn't exist");
186  throw InputMonitorException("Device does not exist");
187  }
188 
189  mChannelPtr = g_io_channel_unix_new(fd);
190 
191  // Read binary data
192  if (g_io_channel_set_encoding(mChannelPtr,
193  NULL,
194  NULL) != G_IO_STATUS_NORMAL) {
195  LOGE("Cannot set encoding for input monitor channel ");
196  throw InputMonitorException("Cannot set encoding");
197  }
198 
199  using namespace std::placeholders;
200  ReadDeviceCallback callback = std::bind(&InputMonitor::readDevice, this, _1);
201  // Add the callback
202  mSourceId = g_io_add_watch_full(mChannelPtr,
203  G_PRIORITY_DEFAULT,
204  G_IO_IN,
205  readDeviceCallback,
206  utils::createCallbackWrapper(callback, mGuard.spawn()),
207  &utils::deleteCallbackWrapper<ReadDeviceCallback>);
208  if (!mSourceId) {
209  LOGE("Cannot add watch on device input file");
210  throw InputMonitorException("Cannot add watch");
211  }
212 }

223 void InputMonitor::readDevice(GIOChannel* gio)
224 {
225  struct input_event ie;
226  gsize nBytesReadTotal = 0;
227  gsize nBytesRequested = sizeof(struct input_event);
228  ScopedGError error;
229 
230  do {
231  gsize nBytesRead = 0;
232  GIOStatus readStatus = g_io_channel_read_chars(gio,
233  &reinterpret_cast<gchar*>(&ie)[nBytesReadTotal],
234  nBytesRequested,
235  &nBytesRead,
236  &error);
237 
238  if (readStatus == G_IO_STATUS_ERROR) {
239  LOGE("Read from input monitor channel failed: " << error->message);
240  return;
241  }
242 
243  nBytesRequested -= nBytesRead;
244  nBytesReadTotal += nBytesRead;
245  } while (nBytesRequested > 0);
246 
247 
248  if (isExpectedEventSequence(ie)) {
249  LOGI("Input monitor detected pattern.");
250  mNotifyCallback();
251  }
252 }

254 bool InputMonitor::isExpectedEventSequence(const struct input_event& ie)
255 {
256  LOGT("Event detected [" << mConfig.device.c_str() << "]:\n"
257  << "\ttime: " << ie.time.tv_sec << "." << ie.time.tv_usec << " sec\n"
258  << "\ttype, code, value: " << ie.type << ", " << ie.code << ", " << ie.value);
259 
260  if (ie.type != EV_KEY
261  || ie.code != mConfig.code
262  || ie.value != KEY_PRESSED) {
263  LOGT("Wrong kind of event");
264  return false;
265  }
266 
267  mEventTimes.push_back(ie.time);
268 
269  if (mEventTimes.size() < static_cast<unsigned int>(mConfig.numberOfEvents)) {
270  LOGT("Event sequence too short");
271  return false;
272  }
273 
274  struct timeval oldest = mEventTimes.front();
275  mEventTimes.pop_front();
276  struct timeval latest = mEventTimes.back();
277 
278  long int secDiff = latest.tv_sec - oldest.tv_sec;
279  if (secDiff >= MAX_TIME_WINDOW_SEC) {
280  LOGT("Time window exceeded");
281  return false;
282  }
283 
284  long int timeDiff = secDiff * 1000L;
285  timeDiff += static_cast<long int>((latest.tv_usec - oldest.tv_usec) / 1000L);
286  if (timeDiff < mConfig.timeWindowMs) {
287  LOGD("Event sequence detected");
288  mEventTimes.clear();
289  return true;
290  }
291 
292  LOGT("Event sequence not detected");
293  return false;
294 }
