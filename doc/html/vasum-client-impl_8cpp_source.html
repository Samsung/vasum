64 void toDict(GVariant* in, VsmArrayString* keys, VsmArrayString* values)
65 {
66  assert(in);
67  assert(keys);
68  assert(values);
69 
70  typedef char* key_type;
71  typedef char* value_type;
72 
73  GVariantIter iter;
74  value_type value;
75  key_type key;
76  gsize size = g_variant_n_children(in);
77  key_type* outk = (key_type*)calloc(size + 1, sizeof(key_type));
78  value_type* outv = (value_type*)calloc(size + 1, sizeof(value_type));
79 
80  g_variant_iter_init(&iter, in);
81  for (int i = 0; g_variant_iter_loop(&iter, "(ss)", &key, &value); i++) {
82  outk[i] = strdup(key);
83  outv[i] = strdup(value);
84  }
85  *keys = outk;
86  *values = outv;
87 }

115 VsmZoneState getZoneState(const char* state)
116 {
117  if (strcmp(state, "STOPPED") == 0) {
118  return STOPPED;
119  } else if (strcmp(state, "STARTING") == 0) {
120  return STARTING;
121  } else if (strcmp(state, "RUNNING") == 0) {
122  return RUNNING;
123  } else if (strcmp(state, "STOPPING") == 0) {
124  return STOPPING;
125  } else if (strcmp(state, "ABORTING") == 0) {
126  return ABORTING;
127  } else if (strcmp(state, "FREEZING") == 0) {
128  return FREEZING;
129  } else if (strcmp(state, "FROZEN") == 0) {
130  return FROZEN;
131  } else if (strcmp(state, "THAWED") == 0) {
132  return THAWED;
133  } else if (strcmp(state, "LOCKED") == 0) {
134  return LOCKED;
135  } else if (strcmp(state, "MAX_STATE") == 0) {
136  return MAX_STATE;
137  } else if (strcmp(state, "ACTIVATING") == 0) {
138  return ACTIVATING;
139  }
140  assert(0 && "UNKNOWN STATE");
141  return (VsmZoneState)-1;
142 }

831 VsmStatus Client::vsm_lookup_netdev_by_name(const char* zone,
832  const char* netdevId,
833  VsmNetdev* netdev) noexcept
834 {
835  using namespace boost::algorithm;
836 
837  assert(zone);
838  assert(netdevId);
839  assert(netdev);
840 
841  NetdevAttrs attrs;
842  VsmStatus ret = getNetdevAttrs(zone, netdevId, attrs);
843  if (ret != VSMCLIENT_SUCCESS) {
844  return ret;
845  }
846 
847  auto it = find_if(attrs.begin(), attrs.end(), [](const tuple<string, string>& entry) {
848  return get<0>(entry) == "type";
849  });
850 
851  VsmNetdevType type;
852  if (it == attrs.end()) {
853  mStatus = Status(VSMCLIENT_OTHER_ERROR, "Can't fetch netdev type");
854  return vsm_get_status();
855  }
856 
857  switch (stoi(get<1>(*it))) {
858  case 1<<0 /*IFF_802_1Q_VLAN*/: type = VSMNETDEV_VETH; break;
859  case 1<<21 /*IFF_MACVLAN*/: type = VSMNETDEV_MACVLAN; break;
860  default:
861  mStatus = Status(VSMCLIENT_CUSTOM_ERROR, "Unknown netdev type: " + get<1>(*it));
862  return vsm_get_status();
863  }
864 
865  *netdev = reinterpret_cast<VsmNetdev>(malloc(sizeof(**netdev)));
866  (*netdev)->name = strdup(zone);
867  (*netdev)->type = type;
868  mStatus = Status();
869  return vsm_get_status();
870 }
