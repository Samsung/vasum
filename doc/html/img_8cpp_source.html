45 // Writes to ret if loop device (provided in loopdev arg) is free to use.
46 // Returns true if check was successful, false if loop device FD was unavailable for some reason.
47 bool isLoopDevFree(const std::string& loopdev, bool& ret)
48 {
49  // initialize
50  ret = false;
51 
52  // open loop device FD
53  int loopFD = ::open(loopdev.c_str(), O_RDWR);
54  if (loopFD < 0) {
55  LOGD("Failed to open loop device descriptor: " << ::strerror(errno));
56  return false;
57  }
58 
59  // if ioctl with LOOP_GET_STATUS fails, device is not assigned and free to use
60  struct loop_info linfo;
61  if (::ioctl(loopFD, LOOP_GET_STATUS, &linfo)) {
62  ret = true;
63  }
64 
65  ::close(loopFD);
66  return true;
67 }

69 bool mountLoop(const std::string& img,
70  const std::string& loopdev,
71  const std::string& path,
72  const std::string& type,
73  unsigned long flags,
74  const std::string& options)
75 {
76  // to mount an image, we need to connect image FD with loop device FD
77  // get image file FD
78  int fileFD = ::open(img.c_str(), O_RDWR);
79  if (fileFD < 0) {
80  LOGD("Failed to open image file descriptor: " << ::strerror(errno));
81  return false;
82  }
83 
84  // get loop device FD
85  int loopFD = ::open(loopdev.c_str(), O_RDWR);
86  if (loopFD < 0) {
87  LOGD("Failed to open loop device descriptor: " << ::strerror(errno));
88  ::close(fileFD);
89  return false;
90  }
91 
92  // set loop device
93  if (::ioctl(loopFD, LOOP_SET_FD, fileFD)) {
94  LOGD("Failed to assign loop device to image: " << ::strerror(errno));
95  ::close(fileFD);
96  ::close(loopFD);
97  return false;
98  }
99 
100  // mount loop device to path
101  if (::mount(loopdev.c_str(), path.c_str(), type.c_str(), flags, options.c_str()) != 0) {
102  LOGD("Mount failed for '" << path << "', options=" << options << ": " << strerror(errno));
103  ::ioctl(loopFD, LOOP_CLR_FD, 0);
104  ::close(fileFD);
105  ::close(loopFD);
106  return false;
107  }
108 
109  ::close(fileFD);
110  ::close(loopFD);
111  return true;
112 }

116 // Finds first available loop device and returns its path through ret.
117 // Returns false if an error occurs, or if all available loop devices are taken.
118 bool getFreeLoopDevice(std::string& ret)
119 {
120  for (unsigned int i = 0; i < 8; ++i) {
121  // build path to loop device
122  const std::string loopdev = LOOP_DEV_PREFIX + std::to_string(i);
123  bool isFree = false;
124 
125  // check if it is free
126  if (!isLoopDevFree(loopdev, isFree)) {
127  LOGD("Failed to check status of " << loopdev);
128  return false;
129  }
130 
131  // if checked loop device is free, we can exit the function and return it
132  if (isFree) {
133  ret = loopdev;
134  return true;
135  }
136  }
137 
138  LOGD("All loop devices are taken.");
139  return false;
140 }
