49 int vsm_recvmsg(int fd, struct msghdr *msg, int flags)
50 {
51  int ret = recvmsg(fd, msg, flags);
52  if (ret < 0) {
53  LOGE("Can't receive message: " + getSystemErrorMessage());
54  } else if (ret == 0 && msg->msg_iov && msg->msg_iov->iov_len > 0) {
55  LOGE("Peer has performed an orderly shutdown");
56  } else if (msg->msg_flags & MSG_TRUNC) {
57  LOGE("Can't receive message: " + getSystemErrorMessage(EMSGSIZE));
58  } else if (msg->msg_flags & MSG_ERRQUEUE) {
59  LOGE("No data was received but an extended error");
60  } else if (msg->msg_flags & MSG_OOB) {
61  LOGE("Internal error (expedited or out-of-band data were received)");
62  } else if (msg->msg_flags & MSG_CTRUNC) {
63  LOGE("Some control data were discarded");
64  } else if (msg->msg_flags & MSG_EOR) {
65  LOGE("End-of-record");
66  } else {
67  // All ok
68  return ret;
69  }
70  throw VasumException("Can't receive netlink message");
71 }

150 std::unique_ptr<std::vector<char>> Netlink::rcv(unsigned int nlmsgSeq)
151 {
152  std::unique_ptr<std::vector<char>> buf(new std::vector<char>());
153 
154  msghdr msg = utils::make_clean<msghdr>();
155  sockaddr_nl nladdr = utils::make_clean<sockaddr_nl>();
156  iovec iov = utils::make_clean<iovec>();
157 
158  msg.msg_name = &nladdr;
159  msg.msg_namelen = sizeof(nladdr);
160  msg.msg_iov = &iov;
161  msg.msg_iovlen = 1;
162  nladdr.nl_family = AF_NETLINK;
163 
164  nlmsghdr* answer;
165  nlmsghdr* lastOk = NULL;
166  size_t offset = 0;
167  do {
168  buf->resize(offset + NLMSG_RCV_GOOD_SIZE);
169  answer = reinterpret_cast<nlmsghdr*>(buf->data() + offset);
170  iov.iov_base = answer;
171  iov.iov_len = buf->size() - offset;
172  unsigned int ret = vsm_recvmsg(mFd, &msg, 0);
173  for (unsigned int len = ret; NLMSG_OK(answer, len); answer = NLMSG_NEXT(answer, len)) {
174  lastOk = answer;
175  if (answer->nlmsg_type == NLMSG_ERROR) {
176  // It is NACK/ACK message
177  nlmsgerr *err = reinterpret_cast<nlmsgerr*>(NLMSG_DATA(answer));
178  if (answer->nlmsg_seq != nlmsgSeq) {
179  throw VasumException("Sending failed: answer message was mismatched");
180  }
181  if (err->error) {
182  throw VasumException("Sending failed: " + getSystemErrorMessage(-err->error));
183  }
184  } else if (answer->nlmsg_type == NLMSG_OVERRUN) {
185  throw VasumException("Sending failed: data lost");
186  }
187  }
188  if (lastOk == NULL) {
189  LOGE("Something went terribly wrong. Check vsm_recvmsg function");
190  throw VasumException("Can't receive data from system");
191  }
192  offset += NLMSG_ALIGN(ret);
193  } while (lastOk->nlmsg_type != NLMSG_DONE && lastOk->nlmsg_flags & NLM_F_MULTI);
194 
195  buf->resize(offset);
196  return buf;
197 }
