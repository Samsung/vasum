136  template<typename T>
137  static void fromGVariant(GVariant* object, std::vector<T>& value)
138  {
139  checkType(object, G_VARIANT_TYPE_ARRAY);
140  GVariantIter iter;
141  g_variant_iter_init(&iter, object);
142  int length = g_variant_iter_n_children(&iter);
143  value.resize(static_cast<size_t>(length));
144  for (int i = 0; i < length; ++i) {
145  auto child = makeUnique(g_variant_iter_next_value(&iter));
146  assert(child);
147  fromGVariant(child.get(), value[static_cast<size_t>(i)]);
148  }
149  }
