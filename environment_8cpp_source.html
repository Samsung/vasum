89  LOGE("Can't receive fd: ret: " << ret << ", flags: " << msg.msg_flags);

198 bool joinToNs(int nsPid, int ns)
199 {
200  auto ins = NAMESPACES.find(ns);
201  if (ins == NAMESPACES.end()) {
202  LOGE("Namespace isn't supported: " << ns);
203  return false;
204  }
205  std::string nsPath = "/proc/" + std::to_string(nsPid) + "/ns/" + ins->second;
206  int nsFd = ::open(nsPath.c_str(), O_RDONLY);
207  if (nsFd == -1) {
208  LOGE("Can't open namesace: " + getSystemErrorMessage());
209  return false;
210  }
211  int ret = setns(nsFd, ins->first);
212  if (ret != 0) {
213  LOGE("Can't set namesace: " + getSystemErrorMessage());
214  close(nsFd);
215  return false;
216  }
217  close(nsFd);
218  return true;
219 }
