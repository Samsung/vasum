102 // hasCap assumes that "set" variable will refer to only one set of capabilities
103 inline bool hasCap(unsigned int cap, const cap_user_data_t data, CapSet set)
104 {
105  // calculate which half of data we need to update
106  int dataInd = 0;
107  if (cap > 31) {
108  dataInd = cap >> 5;
109  cap %= 32;
110  }
111 
112  switch (set) {
113  case CAP_SET_INHERITABLE:
114  return CAP_TO_MASK(cap) & data[dataInd].inheritable ? true : false;
115  case CAP_SET_PERMITTED:
116  return CAP_TO_MASK(cap) & data[dataInd].permitted ? true : false;
117  case CAP_SET_EFFECTIVE:
118  return CAP_TO_MASK(cap) & data[dataInd].effective ? true : false;
119  default:
120  return false;
121  };
122 }

195 bool dropRoot(uid_t uid, gid_t gid, const std::vector<unsigned int>& caps)
196 {
197  ::__user_cap_header_struct header;
198  ::__user_cap_data_struct data[CAP_DATA_ELEMENT_COUNT];
199 
200  // initial setup - equivalent to capng_clear
201  header.version = _LINUX_CAPABILITY_VERSION_3;
202  header.pid = ::getpid();
203  memset(data, 0, CAP_DATA_ELEMENT_COUNT*sizeof(__user_cap_data_struct));
204 
205  // update cap sets - equivalent to capng_update
206  for (const auto cap : caps) {
207  if (!isValidCap(cap)) {
208  LOGE("Capability " << cap << " is invalid.");
209  return false;
210  }
211 
212  addCap(cap, data, CAP_SET_INHERITABLE | CAP_SET_PERMITTED | CAP_SET_EFFECTIVE);
213  }
214 
215  // perform some checks and cap updates
216  bool updatedSetUid, updatedSetGid;
217  // check if we are capable of switching our UID
218  if (hasCap(CAP_SETUID, data, CAP_SET_EFFECTIVE)) {
219  // we want to keep CAP_SETUID after change
220  updatedSetUid = false;
221  } else {
222  // we don't have CAP_SETUID and switch is needed - add SETUID to effective and permitted set
223  updatedSetUid = true;
224  addCap(CAP_SETUID, data, CAP_SET_PERMITTED | CAP_SET_EFFECTIVE);
225  }
226 
227  // do the same routine for CAP_SETGID
228  if (hasCap(CAP_SETGID, data, CAP_SET_EFFECTIVE)) {
229  updatedSetGid = false;
230  } else {
231  updatedSetGid = true;
232  addCap(CAP_SETGID, data, CAP_SET_PERMITTED | CAP_SET_EFFECTIVE);
233  }
234 
235  // we need CAP_SETPCAP as well to clear bounding caps
236  if (!hasCap(CAP_SETPCAP, data, CAP_SET_EFFECTIVE)) {
237  addCap(CAP_SETPCAP, data, CAP_SET_PERMITTED | CAP_SET_EFFECTIVE);
238  }
239 
240  // now we can work - first, use prctl to tell system we want to keep our caps when changing UID
241  if (::prctl(PR_SET_KEEPCAPS, 1, 0, 0, 0)) {
242  LOGE("prctl failed while trying to enable keepcaps: " << strerror(errno));
243  return false;
244  }
245 
246  LOGD("Setting temporary caps to process -" << std::hex << std::setfill('0')
247  << " inh:" << std::setw(8) << data[1].inheritable << std::setw(8) << data[0].inheritable
248  << " prm:" << std::setw(8) << data[1].permitted << std::setw(8) << data[0].permitted
249  << " eff:" << std::setw(8) << data[1].effective << std::setw(8) << data[0].effective);
250 
251  // set our modified caps before UID/GID change
252  if (::capset(&header, data)) {
253  LOGE("capset failed: " << strerror(errno));
254  return false;
255  }
256 
257  // CAP_SETPCAP is available, drop bounding caps
258  for (int i = 0; i <= CAP_LAST_CAP; ++i) {
259  if (::prctl(PR_CAPBSET_DROP, i, 0, 0, 0)) {
260  LOGE("prctl failed while dropping bounding caps: " << strerror(errno));
261  return false;
262  }
263  }
264 
265  // set up GID and UID
266  if (::setresgid(gid, gid, gid)) {
267  LOGE("setresgid failed: " << strerror(errno));
268  return false;
269  }
270  if (::setresuid(uid, uid, uid)) {
271  LOGE("setresuid failed: " << strerror(errno));
272  return false;
273  }
274 
275  // we are after switch now - disable PR_SET_KEEPCAPS
276  if (::prctl(PR_SET_KEEPCAPS, 0, 0, 0, 0)) {
277  LOGE("prctl failed while trying to disable keepcaps: " << strerror(errno));
278  return false;
279  }
280 
281  // disable rendundant caps
282  if (updatedSetUid) {
283  removeCap(CAP_SETUID, data, CAP_SET_PERMITTED | CAP_SET_EFFECTIVE);
284  }
285  if (updatedSetGid) {
286  removeCap(CAP_SETGID, data, CAP_SET_PERMITTED | CAP_SET_EFFECTIVE);
287  }
288  removeCap(CAP_SETPCAP, data, CAP_SET_PERMITTED | CAP_SET_EFFECTIVE);
289 
290  LOGD("Setting final caps to process -" << std::hex << std::setfill('0')
291  << " inh:" << std::setw(8) << data[1].inheritable << std::setw(8) << data[0].inheritable
292  << " prm:" << std::setw(8) << data[1].permitted << std::setw(8) << data[0].permitted
293  << " eff:" << std::setw(8) << data[1].effective << std::setw(8) << data[0].effective);
294 
295  // finally, apply correct caps
296  if (::capset(&header, data)) {
297  LOGE("capset failed: " << strerror(errno));
298  return false;
299  }
300 
301  return true;
302 }

324 bool joinToNs(int nsPid, int ns)
325 {
326  auto ins = NAMESPACES.find(ns);
327  if (ins == NAMESPACES.end()) {
328  LOGE("Namespace isn't supported: " << ns);
329  return false;
330  }
331  std::string nsPath = "/proc/" + std::to_string(nsPid) + "/ns/" + ins->second;
332  int nsFd = ::open(nsPath.c_str(), O_RDONLY);
333  if (nsFd == -1) {
334  LOGE("Can't open namesace: " + getSystemErrorMessage());
335  return false;
336  }
337  int ret = setns(nsFd, ins->first);
338  if (ret != 0) {
339  LOGE("Can't set namesace: " + getSystemErrorMessage());
340  close(nsFd);
341  return false;
342  }
343  close(nsFd);
344  return true;
345 }

347 int passNamespacedFd(int nsPid, int ns, const std::function<int()>& fdFactory)
348 {
349  int fds[2];
350  int ret = socketpair(PF_LOCAL, SOCK_RAW, 0, fds);
351  if (ret == -1) {
352  LOGE("Can't create socket pair: " << getSystemErrorMessage());
353  return -1;
354  }
355  bool success = executeAndWait([&, fds, nsPid, ns]() {
356  close(fds[0]);
357 
358  int fd = -1;
359  if (joinToNs(nsPid, ns)) {
360  fd = fdFactory();
361  }
362  if (fd == -1) {
363  close(fds[1]);
364  _exit(EXIT_FAILURE);
365  }
366  LOGT("FD pass, send: " << fd);
367  fdSend(fds[1], fd);
368  close(fds[1]);
369  close(fd);
370  });
371 
372  close(fds[1]);
373  int fd = -1;
374  if (success) {
375  fd = fdRecv(fds[0]);
376  }
377  close(fds[0]);
378  LOGT("FD pass, rcv: " << fd);
379  return fd;
380 }
