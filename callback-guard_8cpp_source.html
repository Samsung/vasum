36 // Reference counting class like shared_ptr but with the ability to wait for it.
37 class CallbackGuard::SharedState {
38 public:

45  }

60  bool wait(const unsigned int timeoutMs)
61  {
62  std::unique_lock<std::mutex> lock(mMutex);
63  return mEmptyCondition.wait_for(lock,
64  std::chrono::milliseconds(timeoutMs),
65  [this] {return mCounter == 0;});
66  }
67 private:
