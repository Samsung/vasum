101 bool ZoneConnection::waitForNameAndSetCallback(const unsigned int timeoutMs, const OnNameLostCallback& callback)
102 {
103  std::unique_lock<std::mutex> lock(mNameMutex);
104  mNameCondition.wait_for(lock,
105  std::chrono::milliseconds(timeoutMs),
106  [this] {
107  return mNameAcquired || mNameLost;
108  });
109  if(mNameAcquired) {
110  mOnNameLostCallback = callback;
111  }
112 
113  return mNameAcquired;
114 }

156 void ZoneConnection::onMessageCall(const std::string& objectPath,
157  const std::string& interface,
158  const std::string& methodName,
159  GVariant* parameters,
160  dbus::MethodResultBuilder::Pointer result)
161 {
162  if (objectPath != api::zone::OBJECT_PATH || interface != api::zone::INTERFACE) {
163  return;
164  }
165 
166  if (methodName == api::zone::METHOD_NOTIFY_ACTIVE_ZONE) {
167  const gchar* application = NULL;
168  const gchar* message = NULL;
169  g_variant_get(parameters, "(&s&s)", &application, &message);
170  if (mNotifyActiveZoneCallback) {
171  auto rb = std::make_shared<api::DbusMethodResultBuilder<api::Void>>(result);
172  mNotifyActiveZoneCallback(application, message, rb);
173  }
174  }
175 
176  if (methodName == api::zone::METHOD_FILE_MOVE_REQUEST) {
177  const gchar* destination = NULL;
178  const gchar* path = NULL;
179  g_variant_get(parameters, "(&s&s)", &destination, &path);
180  if (mFileMoveCallback) {
181  auto rb = std::make_shared<api::DbusMethodResultBuilder<api::FileMoveRequestStatus>>(result);
182  mFileMoveCallback(destination, path, rb);
183  }
184  }
185 
186  if (methodName == api::METHOD_PROXY_CALL) {
187  const gchar* target = NULL;
188  const gchar* targetBusName = NULL;
189  const gchar* targetObjectPath = NULL;
190  const gchar* targetInterface = NULL;
191  const gchar* targetMethod = NULL;
192  GVariant* rawArgs = NULL;
193  g_variant_get(parameters,
194  "(&s&s&s&s&sv)",
195  &target,
196  &targetBusName,
197  &targetObjectPath,
198  &targetInterface,
199  &targetMethod,
200  &rawArgs);
201  dbus::GVariantPtr args(rawArgs, g_variant_unref);
202 
203  if (mProxyCallCallback) {
204  mProxyCallCallback(target,
205  targetBusName,
206  targetObjectPath,
207  targetInterface,
208  targetMethod,
209  args.get(),
210  result);
211  }
212  }
213 }
