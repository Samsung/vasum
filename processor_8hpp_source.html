504 void Processor::setMethodHandlerInternal(const MethodID methodID,
505  const typename MethodHandler<SentDataType, ReceivedDataType>::type& method)
506 {
507  MethodHandlers methodCall;
508 
509  methodCall.parse = [](const int fd)->std::shared_ptr<void> {
510  std::shared_ptr<ReceivedDataType> data(new ReceivedDataType());
511  config::loadFromFD<ReceivedDataType>(fd, *data);
512  return data;
513  };
514 
515  methodCall.serialize = [](const int fd, std::shared_ptr<void>& data)->void {
516  config::saveToFD<SentDataType>(fd, *std::static_pointer_cast<SentDataType>(data));
517  };
518 
519  methodCall.method = [method](const PeerID peerID, std::shared_ptr<void>& data, MethodResult::Pointer && methodResult) {
520  std::shared_ptr<ReceivedDataType> tmpData = std::static_pointer_cast<ReceivedDataType>(data);
521  method(peerID, tmpData, std::forward<MethodResult::Pointer>(methodResult));
522  };
523 
524  mMethodsCallbacks[methodID] = std::make_shared<MethodHandlers>(std::move(methodCall));
525 }

570 template<typename ReceivedDataType>
571 void Processor::setSignalHandler(const MethodID methodID,
572  const typename SignalHandler<ReceivedDataType>::type& handler)
573 {
574  if (methodID == RETURN_METHOD_ID || methodID == REGISTER_SIGNAL_METHOD_ID) {
575  LOGE(mLogPrefix + "Forbidden methodID: " << methodID);
576  throw IPCException("Forbidden methodID: " + std::to_string(methodID));
577  }
578 
579  std::shared_ptr<RegisterSignalsProtocolMessage> data;
580 
581  {
582  Lock lock(mStateMutex);
583 
584  // Andd the signal handler:
585  if (mMethodsCallbacks.count(methodID)) {
586  LOGE(mLogPrefix + "MethodID used by a method: " << methodID);
587  throw IPCException("MethodID used by a method: " + std::to_string(methodID));
588  }
589 
590  setSignalHandlerInternal<ReceivedDataType>(methodID, handler);

592  // Broadcast the new signal:

627 template<typename SentDataType, typename ReceivedDataType>
628 std::shared_ptr<ReceivedDataType> Processor::callSync(const MethodID methodID,
629  const PeerID peerID,
630  const std::shared_ptr<SentDataType>& data,
631  unsigned int timeoutMS)
632 {
633  Result<ReceivedDataType> result;
634  std::condition_variable cv;
635 
636  auto process = [&result, &cv](const Result<ReceivedDataType> && r) {
637  // This is called under lock(mStateMutex)
638  result = std::move(r);
639  cv.notify_all();
640  };
641 
642  Lock lock(mStateMutex);
643  MessageID messageID = callAsyncInternal<SentDataType, ReceivedDataType>(methodID,
644  peerID,
645  data,
646  process);
647 
648  auto isResultInitialized = [&result]() {
649  return result.isValid();
650  };
651 
652  LOGT(mLogPrefix + "Waiting for the response...");
653  //In the case of too large sending time response can be received far after timeoutMS but
654  //before this thread wakes up and before predicate check (there will by no timeout exception)
655  if (!cv.wait_for(lock, std::chrono::milliseconds(timeoutMS), isResultInitialized)) {
656  LOGW(mLogPrefix + "Probably a timeout in callSync. Checking...");
657 
658  // Call isn't sent or call is sent but there is no reply
659  bool isTimeout = mRequestQueue.removeIf([messageID](Request & request) {
660  return request.requestID == Event::METHOD &&
661  request.get<MethodRequest>()->messageID == messageID;
662  })
663  || 1 == mReturnCallbacks.erase(messageID);
664 
665  if (isTimeout) {
666  LOGE(mLogPrefix + "Function call timeout; methodID: " << methodID);
667  removePeerSyncInternal(peerID, lock);
668  throw IPCTimeoutException("Function call timeout; methodID: " + std::to_string(methodID));
669  } else {
670  LOGW(mLogPrefix + "Timeout started during the return value processing, so wait for it to finish");
671  if (!cv.wait_for(lock, std::chrono::milliseconds(timeoutMS), isResultInitialized)) {
672  LOGE(mLogPrefix + "Function call timeout; methodID: " << methodID);
673  throw IPCTimeoutException("Function call timeout; methodID: " + std::to_string(methodID));
674  }
675  }
676  }
677 
678  return result.get();
679 }

690 template<typename SentDataType>
691 void Processor::signal(const MethodID methodID,
692  const std::shared_ptr<SentDataType>& data)
693 {
694  Lock lock(mStateMutex);
695  const auto it = mSignalsPeers.find(methodID);
696  if (it == mSignalsPeers.end()) {
697  LOGW(mLogPrefix + "No peer is handling signal with methodID: " << methodID);
698  return;
