74  // create visitor for field object (visitable object)
75  FromKVJsonVisitor(const FromKVJsonVisitor& v, const std::string& name, const bool isUnion) :
76  mStore(v.mStore),
77  mKeyPrefix(key(v.mKeyPrefix, name)),
78  mIsUnion(isUnion || v.mIsUnion)
79  {
80  json_object* object = nullptr;
81  if (v.mObject && !json_object_object_get_ex(v.mObject, name.c_str(), &object)) {
82  if (!mIsUnion)
83  throw ConfigException("Missing json field " + mKeyPrefix);
84  }
85  mObject = object ? json_object_get(object) : nullptr;
86  }

88  // create visitor for vector i-th element (visitable object)
89  FromKVJsonVisitor(const FromKVJsonVisitor& v, int i, const bool isUnion) :
90  mStore(v.mStore),
91  mKeyPrefix(key(v.mKeyPrefix, std::to_string(i))),
92  mIsUnion(isUnion || v.mIsUnion)
93  {
94  json_object* object = nullptr;
95  if (v.mObject) {
96  object = json_object_array_get_idx(v.mObject, i);
97  checkType(object, json_type_object);
98  }
99  mObject = object ? json_object_get(object) : nullptr;

102  template<typename T, typename std::enable_if<!isVisitable<T>::value, int>::type = 0>
103  void getValue(const std::string& name, T& t)
104  {
105  std::string k = key(mKeyPrefix, name);
106  if (mStore.exists(k)) {
107  t = mStore.get<T>(k);
108  }
109  else {
110  json_object* object = nullptr;
111  if (mObject) {
112  json_object_object_get_ex(mObject, name.c_str(), &object);
113  }
114  if (!object) {
115  throw ConfigException("Missing json field " + key(mKeyPrefix, name));
116  }
117  fromJsonObject(object, t);
118  }
119  }

146  template<typename T>
147  void getValue(const std::string& name, std::vector<T>& value)
148  {
149  json_object* object = nullptr;
150  if (mObject && json_object_object_get_ex(mObject, name.c_str(), &object)) {
151  checkType(object, json_type_array);
152  }
153 
154  std::string k = key(mKeyPrefix, name);
155  int length = getArraySize(k, object);
156  if (length < 0) {
157  throw ConfigException("Missing array length " + k);
158  }
159  value.resize(static_cast<size_t>(length));
160  FromKVJsonVisitor visitor(*this, name, false);
161  if (mStore.exists(k)) {
162  json_object_put(visitor.mObject);
163  visitor.mObject = nullptr;
164  }
165  for (int i = 0; i < length; ++i) {
166  visitor.getValue(i, value[i]);
167  }
168  }

223  static void fromJsonObject(json_object* object, int& value)
224  {
225  checkType(object, json_type_int);
226  std::int64_t value64 = json_object_get_int64(object);
