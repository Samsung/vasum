169 template<typename RequestIdType>

171 bool RequestQueue<RequestIdType>::removeIf(Predicate predicate)
172 {
173  Lock lock(mStateMutex);
174  auto it = std::find_if(mRequests.begin(), mRequests.end(), predicate);
175  if (it == mRequests.end()) {
176  return false;
177  }
178 
179  do {
180  it = mRequests.erase(it);
181  it = std::find_if(it, mRequests.end(), predicate);
182  } while (it != mRequests.end());
183 
184  return true;
