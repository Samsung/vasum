200  mConnection.reset(new ZoneConnection(mDbusAddress,

358 void Zone::reconnectHandler()
359 {
360  {
361  Lock lock(mReconnectMutex);
362  disconnect();
363  }
364 
365  for (int i = 0; i < RECONNECT_RETRIES; ++i) {
366  // This sleeps even before the first try to give DBUS some time to come back up
367  std::this_thread::sleep_for(std::chrono::milliseconds(RECONNECT_DELAY));
368 
369  Lock lock(mReconnectMutex);
370  if (isStopped()) {
371  LOGI(getId() << ": Has stopped, nothing to reconnect to, bailing out");
372  return;
373  }
374 
375  try {
376  LOGT(getId() << ": Reconnect try " << i + 1);
377  connect();
378  LOGI(getId() << ": Reconnected");
379  return;
380  } catch (VasumException&) {
381  LOGT(getId() << ": Reconnect try " << i + 1 << " has been unsuccessful");
382  }
383  }
384 
385  LOGE(getId() << ": Reconnecting to the DBUS has failed, stopping the zone");
386  stop(false);
387 }
