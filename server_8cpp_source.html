154 bool Server::checkEnvironment()
155 {
156  // check kernel
157  struct utsname u;
158  int version, major, minor;
159  ::uname(&u);
160  version = major = minor = 0;
161  ::sscanf(u.release, "%d.%d.%d", &version, &major, &minor);
162  if (version < 2 || (version == 2 && major < 6) || (version == 2 && major == 6 && minor < 29)) {
163  // control-group functionality was merged into kernel version 2.6.24 in 2007 (wikipedia)
164  // namespace support begins from kernels 2.4.19(mnt), 2.6.19(ns,uts,ipc), 2.6.24(pid), 2.6.29(net)
165  // namespace for usr from kernel 3.8(usr) - not used by vasum
166  std::cout << "kernel is old ver=" << u.release << ", run vasum-check-config" << std::endl;
167  return false;
168  }
169  else
170  std::cout << "kernel " << u.release << " [OK]" << std::endl;
171 
172  // check lxc (TODO check if running on broken ABI version)
173  if (::strcmp(lxc_get_version(), LXC_VERSION)!=0) {
174  // versions that matters:
175  // 1.1.0 added function ptr 'in-the-middle' destroy_with_snapshots, snapshot_destroy_all (breaks ABI)
176  // 1.1.2 added function ptr 'append' attach_interface,detach_interface,checkpoint,restore (safe for ABI)
177  std::cout << "LXC version not match, compiled for " << LXC_VERSION << ", installed " << lxc_get_version() << std::endl;
178  return false;
179  }
180  else
181  std::cout << "LXC version " << lxc_get_version() << " [OK]" << std::endl;
182 
183  // check cgroups (and its subsystems?)
184  std::string cgroupCheck = "/sys/fs/cgroup";
185  int fd = ::open(cgroupCheck.c_str(), O_RDONLY);
186  if (fd == -1) {
187  std::cout << "no cgroups support (can't access " << cgroupCheck << "), run vasum-check-config" << std::endl;
188  return false;
189  }
190 
191  bool err = false;
192  std::vector<std::string> cgroupSubsCheck = {"cpu", "cpuset", "memory"};
193  for (std::string f : cgroupSubsCheck) {
194  if (::faccessat(fd, f.c_str(), R_OK|X_OK, 0) == -1) {
195  std::cout << "no cgroups support (can't access " << cgroupCheck << "/" << f << ")" << std::endl;
196  err=true;
197  }
198  }
199  ::close(fd);
200  if (err) {
201  std::cout << "cgroups problem, run vasum-check-config" << std::endl;
202  return false;
203  }
204  else
205  std::cout << "cgroups support " << " [OK]" << std::endl;
206 
207  // check namespaces
208  std::string nsCheck = "/proc/self/ns";
209  if (::access(nsCheck.c_str(), R_OK|X_OK) == -1) {
210  std::cout << "no namespace support (can't access " << nsCheck << "), run vasum-check-config" << std::endl;
211  return false;
212  }
213  else
214  std::cout << "namespaces support " << " [OK]" << std::endl;
215  ::close(fd);
216 
217  return true;
218 }

221 bool Server::prepareEnvironment(const std::string& configPath, bool runAsRoot)
222 {
223  namespace fs = boost::filesystem;
224 
225  // TODO: currently this config is loaded twice: here and in ZonesManager
226  ZonesManagerConfig config;
227  config::loadFromJsonFile(configPath, config);
228 
229  struct passwd* pwd = ::getpwnam(VASUM_USER);
230  if (pwd == NULL) {
231  LOGE("getpwnam failed to find user '" << VASUM_USER << "'");
232  return false;
233  }
234  uid_t uid = pwd->pw_uid;
235  gid_t gid = pwd->pw_gid;
236  LOGD("vasum UID = " << uid << ", GID = " << gid);
237 
238  // create directory for dbus socket (if needed)
239  if (!config.runMountPointPrefix.empty()) {
240  if (!utils::createDir(config.runMountPointPrefix, uid, gid,
241  fs::perms::owner_all |
242  fs::perms::group_read | fs::perms::group_exe |
243  fs::perms::others_read | fs::perms::others_exe)) {
244  return false;
245  }
246  }
247 
248  // Omit supplementaty group setup and root drop if the user is already switched.
249  // This situation will happen during daemon update triggered by SIGUSR1.
250  if (!runAsRoot && geteuid() == uid) {
251  return true;
252  }
253 
254  // INPUT_EVENT_GROUP provides access to /dev/input/event* devices used by InputMonitor.
255  // DISK_GROUP provides access to /dev/loop* devices, needed when adding new zone to copy
256  // zones image
257  if (!utils::setSuppGroups({INPUT_EVENT_GROUP, DISK_GROUP, TTY_GROUP})) {
258  return false;
259  }
260 
261  // CAP_SYS_ADMIN allows to mount tmpfs' for dbus communication at the runtime.
262  // NOTE: CAP_MAC_OVERRIDE is temporary and must be removed when "smack namespace"
263  // is introduced. The capability is needed to allow modify SMACK labels of
264  // "/var/run/zones/<zone>/run" mount point.
265  // CAP_SYS_TTY_CONFIG is needed to activate virtual terminals through ioctl calls
266  // CAP_CHOWN is needed when creating new zone from image to set owner/group for each file,
267  // directory or symlink
268  // CAP_SETUID is needed to launch specific funtions as root (see environment.cpp)
269  return (runAsRoot || utils::dropRoot(uid, gid, {CAP_SYS_ADMIN,
270  CAP_MAC_OVERRIDE,
271  CAP_SYS_TTY_CONFIG,
272  CAP_CHOWN,
273  CAP_SETUID}));
274 }
