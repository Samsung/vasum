141 bool Server::prepareEnvironment(const std::string& configPath, bool runAsRoot)
142 {
143  namespace fs = boost::filesystem;
144 
145  // TODO: currently this config is loaded twice: here and in ZonesManager
146  ZonesManagerConfig config;
147  config::loadFromJsonFile(configPath, config);
148 
149  struct passwd* pwd = ::getpwnam(VASUM_USER);
150  if (pwd == NULL) {
151  LOGE("getpwnam failed to find user '" << VASUM_USER << "'");
152  return false;
153  }
154  uid_t uid = pwd->pw_uid;
155  gid_t gid = pwd->pw_gid;
156  LOGD("vasum UID = " << uid << ", GID = " << gid);
157 
158  // create directory for dbus socket (if needed)
159  if (!config.runMountPointPrefix.empty()) {
160  if (!utils::createDir(config.runMountPointPrefix, uid, gid,
161  fs::perms::owner_all |
162  fs::perms::group_read | fs::perms::group_exe |
163  fs::perms::others_read | fs::perms::others_exe)) {
164  return false;
165  }
166  }
167 
168  // Omit supplementaty group setup and root drop if the user is already switched.
169  // This situation will happen during daemon update triggered by SIGUSR1.
170  if (!runAsRoot && geteuid() == uid) {
171  return true;
172  }
173 
174  // INPUT_EVENT_GROUP provides access to /dev/input/event* devices used by InputMonitor.
175  // DISK_GROUP provides access to /dev/loop* devices, needed when adding new zone to copy
176  // zones image
177  if (!utils::setSuppGroups({INPUT_EVENT_GROUP, DISK_GROUP, TTY_GROUP})) {
178  return false;
179  }
180 
181  // CAP_SYS_ADMIN allows to mount tmpfs' for dbus communication at the runtime.
182  // NOTE: CAP_MAC_OVERRIDE is temporary and must be removed when "smack namespace"
183  // is introduced. The capability is needed to allow modify SMACK labels of
184  // "/var/run/zones/<zone>/run" mount point.
185  // CAP_SYS_TTY_CONFIG is needed to activate virtual terminals through ioctl calls
186  // CAP_CHOWN is needed when creating new zone from image to set owner/group for each file,
187  // directory or symlink
188  // CAP_SETUID is needed to launch specific funtions as root (see environment.cpp)
189  return (runAsRoot || utils::dropRoot(uid, gid, {CAP_SYS_ADMIN,
190  CAP_MAC_OVERRIDE,
191  CAP_SYS_TTY_CONFIG,
192  CAP_CHOWN,
193  CAP_SETUID}));
194 }
