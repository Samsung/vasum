50 HostDbusConnection::HostDbusConnection(ZonesManager* zonesManagerPtr)
51  : mNameAcquired(false)
52  , mNameLost(false)
53  , mZonesManagerPtr(zonesManagerPtr)
54 {
55  LOGT("Connecting to host system DBUS");
56  mDbusConnection = dbus::DbusConnection::createSystem();
57 
58  LOGT("Setting DBUS name");
59  mDbusConnection->setName(api::dbus::BUS_NAME,
60  std::bind(&HostDbusConnection::onNameAcquired, this),
61  std::bind(&HostDbusConnection::onNameLost, this));
62 
63  if (!waitForName(NAME_ACQUIRED_TIMEOUT)) {
64  LOGE("Could not acquire dbus name: " << api::dbus::BUS_NAME);
65  throw HostConnectionException("Could not acquire dbus name: " + api::dbus::BUS_NAME);
66  }
67 
68  LOGT("Registering DBUS interface");
69  using namespace std::placeholders;
70  mDbusConnection->registerObject(api::dbus::OBJECT_PATH,
71  api::dbus::DEFINITION,
72  std::bind(&HostDbusConnection::onMessageCall,
73  this, _1, _2, _3, _4, _5),
74  std::bind(&HostDbusConnection::onClientVanished,
75  this, _1));
76 
77  mSubscriptionId = mDbusConnection->signalSubscribe(std::bind(&HostDbusConnection::onSignalCall,
78  this, _1, _2, _3, _4, _5),
79  std::string(),
80  api::dbus::INTERFACE);
81  LOGD("Connected");
82 }

89 bool HostDbusConnection::waitForName(const unsigned int timeoutMs)
90 {
91  std::unique_lock<std::mutex> lock(mNameMutex);
92  mNameCondition.wait_for(lock,
93  std::chrono::milliseconds(timeoutMs),
94  [this] {
95  return mNameAcquired || mNameLost;
96  });
97 
98  return mNameAcquired;
99 }
