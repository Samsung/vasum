71 std::string ZoneProvision::declareProvision(ZoneProvisioningConfig::Provision&& provision)
72 {
73  std::string id = getId(provision);
74  auto it = std::find_if(mProvisioningConfig.provisions.begin(),
75  mProvisioningConfig.provisions.end(),
76  [&](const ZoneProvisioningConfig::Provision& existingProvision) {
77  return getId(existingProvision) == id;
78  });
79  if (it != mProvisioningConfig.provisions.end()) {
80  LOGE("Can't add provision. It already exists: " << id);
81  throw UtilsException("Provision already exists");
82  }
83  mProvisioningConfig.provisions.push_back(std::move(provision));
84  saveProvisioningConfig();
85  return id;

120 void ZoneProvision::start() noexcept
121 {
122  for (const auto& provision : mProvisioningConfig.provisions) {
123  try {
124  if (provision.is<ZoneProvisioningConfig::File>()) {
125  file(provision.as<ZoneProvisioningConfig::File>());
126  } else if (provision.is<ZoneProvisioningConfig::Mount>()) {
127  mount(provision.as<ZoneProvisioningConfig::Mount>());
128  } else if (provision.is<ZoneProvisioningConfig::Link>()) {
129  link(provision.as<ZoneProvisioningConfig::Link>());
130  }
131  // mProvisioned must be FILO
132  mProvisioned.push_front(provision);
133  } catch (const std::exception& ex) {
134  LOGE("Provsion error: " << ex.what());
135  }
136  }
137 }

180 void ZoneProvision::file(const ZoneProvisioningConfig::File& config)
181 {
182  bool ret = false;
183  const fs::path hostPath = fs::path(mRootPath) / fs::path(config.path);
184  switch (config.type) {
185  case VSMFILE_DIRECTORY:
186  ret = utils::createDirs(hostPath.string(), config.mode);
187  if (!ret) {
188  throw UtilsException("Can't create dir: " + hostPath.string());
189  }
190  break;
191 
192  case VSMFILE_FIFO:
193  ret = utils::createFifo(hostPath.string(), config.mode);
194  if (!ret) {
195  throw UtilsException("Failed to make fifo: " + config.path);
196  }
197  break;
198 
199  case VSMFILE_REGULAR:
200  if ((config.flags & O_CREAT)) {
201  ret = utils::createFile(hostPath.string(), config.flags, config.mode);
202  if (!ret) {
203  throw UtilsException("Failed to create file: " + config.path);
204  }
205  } else {
206  ret = utils::copyFile(config.path, hostPath.string());
207  if (!ret) {
208  throw UtilsException("Failed to copy file: " + config.path);
209  }
210  }
211  break;
212  }
213 }

237 void ZoneProvision::link(const ZoneProvisioningConfig::Link& config)
238 {
239  const std::string srcHostPath = fs::path(config.source).normalize().string();
240  for (const std::string& prefix : mValidLinkPrefixes) {
241  if (prefix.length() <= srcHostPath.length()
242  && srcHostPath.compare(0, prefix.length(), prefix) == 0) {
243 
244  const fs::path destHostPath = fs::path(mRootPath) / fs::path(config.target);
245  bool ret = utils::createLink(srcHostPath, destHostPath.string());
246  if (!ret) {
247  throw UtilsException("Failed to create hard link: " + config.source);
248  }
249  return;
250  }
251  }
252  LOGE("Failed to create hard link: path=host: "
253  << srcHostPath
254  << ", msg: Path prefix is not valid path");
255  throw UtilsException("Failed to hard link: path prefix is not valid");
256 }
