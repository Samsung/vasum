86 static inline bool operator==(const InetAddr& a, const InetAddr& b) {
87  if (a.type == b.type && a.prefix == b.prefix) {
88  if (a.type == InetAddrType::IPV4) {
89  return ::memcmp(&a.addr.ipv4, &b.addr.ipv4, sizeof(a.addr.ipv4)) == 0;
90  }
91  if (a.type == InetAddrType::IPV6) {
92  return ::memcmp(&a.addr.ipv6, &b.addr.ipv6, sizeof(a.addr.ipv6)) == 0;
93  }
94  }
95  return false;
96 }
