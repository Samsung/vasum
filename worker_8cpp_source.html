114  void workerProc()
115  {
116  LOGT("Worker thread started");
117  for (;;) {
118  // wait for a task
119  GroupID groupID;
120  {
121  Lock lock(mMutex);
122  mAddedCondition.wait(lock, [this] {
123  return !mTaskQueue.empty() || mEnding;
124  });
125  if (mTaskQueue.empty()) {
126  break;
127  }
128  TaskInfo taskInfo = std::move(mTaskQueue.front());
129  mTaskQueue.pop_front();
130 
131  lock.unlock();
132 
133  // execute
134  execute(taskInfo);
135  groupID = taskInfo.groupID;
136  }
137  // remove from queue
138  {
139  Lock lock(mMutex);
140  if (mGroupCounter.decrement(groupID) == 0) {
141  mEmptyGroupCondition.notify_all();
142  }
143  }
144  }
145  LOGT("Worker thread exited");
146  }
