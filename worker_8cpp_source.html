113  void workerProc()
114  {
115  LOGT("Worker thread started");
116  for (;;) {
117  // wait for a task
118  GroupID groupID;
119  {
120  Lock lock(mMutex);
121  mAddedCondition.wait(lock, [this] {
122  return !mTaskQueue.empty() || mEnding;
123  });
124  if (mTaskQueue.empty()) {
125  break;
126  }
127  TaskInfo taskInfo = std::move(mTaskQueue.front());
128  mTaskQueue.pop_front();
129 
130  lock.unlock();
131 
132  // execute
133  execute(taskInfo);
134  groupID = taskInfo.groupID;
135  }
136  // remove from queue
137  {
138  Lock lock(mMutex);
139  if (mGroupCounter.decrement(groupID) == 0) {
140  mEmptyGroupCondition.notify_all();
141  }
142  }
143  }
144  LOGT("Worker thread exited");
145  }
